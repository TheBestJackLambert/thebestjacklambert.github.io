# Forward Kinematics - Python import math def forward_kinematics(L1, L2, theta1, theta2): """Calculate end-effector position""" x = L1 * math.cos(theta1) + L2 * math.cos(theta1 + theta2) y = L1 * math.sin(theta1) + L2 * math.sin(theta1 + theta2) return x, y def inverse_kinematics(L1, L2, x, y): """Calculate joint angles from target position""" r_sq = x**2 + y**2 cos_theta2 = (r_sq - L1**2 - L2**2) / (2 * L1 * L2) theta2 = math.acos(cos_theta2) theta1 = math.atan2(y, x) - math.atan2(L2*math.sin(theta2), L1+L2*math.cos(theta2)) return theta1, theta2
// Forward Kinematics - C #include <math.h> typedef struct { float x, y; } Point; Point forward_kinematics(float L1, float L2, float t1, float t2) { Point p; p.x = L1 * cos(t1) + L2 * cos(t1 + t2); p.y = L1 * sin(t1) + L2 * sin(t1 + t2); return p; }
// Inverse Kinematics - RobotC float computeTheta2(float x, float y, float L1, float L2) { float r_sq = x*x + y*y; float cos_t2 = (r_sq - L1*L1 - L2*L2) / (2*L1*L2); return acos(cos_t2); } float computeTheta1(float x, float y, float L1, float L2, float t2) { return atan2(y, x) - atan2(L2*sin(t2), L1+L2*cos(t2)); }
