trig.py — Custom Taylor Series Trig Library 
# pi constant
pi = 3.14159265359159

# factorial
def fac(x):
  y = 1
  for i in range(1, x+1):
    y *= i
  return y

# mod along pi
def wrap(a):
  a = (a + pi) % (2 * pi) - pi
  return a

# cosine — Taylor series, 25 terms
def cos(x):
  x = wrap(x)
  y = 0
  for i in range(25):
    y += 1/(fac(2*i)) * x ** (2*i) * (-1)**i
  return y

# sine — Taylor series, 25 terms
def sin(x):
  x = wrap(x)
  y = 0
  for i in range(25):
    y += 1/(fac(1 + 2*i)) * x ** (2*i + 1) * (-1)**i
  return y

# signal generators
def sine(x, freq, N, phase, amp):
    return sin(x * 2 * pi * freq / N + phase) * amp

def cosine(x, freq, N, phase, amp):
    return cos(x * 2 * pi * freq / N + phase) * amp
signals.py — DFT Algorithm 
import trig

def DFT(x):
    N = len(x)
    smp = 1  # sampling rate = 1 for simplicity

    # frequency strength lists
    freqa = []
    freqb = []
    stren = []
    phase = []

    # cycles the frequencies
    for i in range(N):
        a = 0
        b = 0

        # cycles the points
        for j in range(N):
            a += x[j] * trig.sin(2 * trig.pi * j * i * 1/N)
            b += x[j] * trig.cos(2 * trig.pi * j * i * 1/N)

        freqa.append(a)
        freqb.append(b)

    # computes magnitude and phase
    for i in range(N):
        stren.append(((freqa[i])**2 + (freqb[i])**2) ** .5)
        phase.append(trig.arctan(freqa[i], freqb[i]))

    return stren, phase

// HOW THE DFT WORKS

The Discrete Fourier Transform correlates the input signal with sinusoids at each frequency bin. For each frequency k, we multiply every sample by a complex exponential at that frequency and sum the results:

// THE_EQUATION

X[k] = Σ x[n] · e^(-j·2π·k·n/N)

     = Σ x[n] · [cos(2πkn/N) - j·sin(2πkn/N)]

for k = 0, 1, ..., N-1

The magnitude |X[k]| tells us how much of frequency k is present in the signal. Try adding a 5 Hz sine in the interactive above — you'll see a single spike at bin 5.

// CONVOLUTION

y[n] = Σ h[k] · x[n - k]

Output = Input ∗ Impulse Response
Convolution in time = Multiplication in frequency
